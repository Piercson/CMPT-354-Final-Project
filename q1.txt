select C.id, C.title
from Call C
where C.deadline >= USER_MONTH_INPUT AND C.status = 'open' AND EXISTS ( select *
									from proposal P
									where P.callid = C.id AND P.appstatus = 'submitted' AND (P.requestedamount > 20,000 OR 10 < ( 	select count(col.researcherid)
					 																from collaborator col
					 																group by col.researcherid
					 																having col.proposalid = p.id) ));
